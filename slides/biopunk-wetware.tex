\begin{frame}{The one-slide thesis}
  \begin{block}{Chi.Bio and OpenEvo look like competitors. They aren't.}
  \begin{itemize}
    \item \ac{Chi.Bio is an instrument} --- measures a culture richly, in situ, while holding it in a defined state.
    \item \ac{OpenEvo is an actuator} --- pushes a culture through selection, unattended, for weeks.
  \end{itemize}
  \end{block}
  \vspace{0.5em}
  Both cost $\sim$\$300. Both are open. But a machine built for one purpose is
  \emph{measurably worse} at the other. \\[0.4em]
  \textbf{The field serves three purposes, not one contest.}
\end{frame}

\begin{frame}{Three purposes}
  \begin{columns}[T]
    \column{0.33\textwidth}
      \textbf{\textcolor{accent}{proto -- Characterise}}\\[0.3em]
      Hold a defined state; measure richly in situ; close feedback loops.\\[0.4em]
      \emph{Champion:} \ac{Chi.Bio}
    \column{0.33\textwidth}
      \textbf{\textcolor{violet}{devo -- Evolve}}\\[0.3em]
      Weeks of unattended selection; produce \emph{archived, replicated} evidence.\\[0.4em]
      \emph{Champion:} \bd{none yet}
    \column{0.33\textwidth}
      \textbf{\textcolor{warn}{bior -- Scale out}}\\[0.3em]
      Many reliable reactors now; turnkey; supported.\\[0.4em]
      \emph{Champions:} Pioreactor, eVOLVER
  \end{columns}
  \vspace{0.6em}
  \begin{center}
  \textcolor{accent}{\textbf{proto}} $\rightarrow$ \textcolor{violet}{\textbf{devo}}
  $\rightarrow$ \textcolor{warn}{\textbf{bior}} \quad --- a lifecycle: prototype, evolve, produce.
  \end{center}
  \vspace{0.3em}
  \begin{center}\small
  Rich-\&-few vs.\ cheap-\&-many \quad$|$\quad non-contact vs.\ electrodes-OK \quad$|$\quad
  steady-state vs.\ run-to-stationary
  \end{center}
\end{frame}

\begin{frame}{OpenEvo: credit where due}
  A real contribution --- these are genuine strengths:
  \begin{itemize}
    \item \gd{Peak-to-peak growth readout} tied to hardware dilution events (no local-maxima noise).
    \item \gd{940\,nm OD} avoids optogenetic \& ambient overlap; $R^2\!\ge\!0.98$ to $\mathrm{OD}_{600}=6$.
    \item \gd{Closed autoclavable vessel} --- better sterility than Chi.Bio's open tube.
    \item \gd{4{,}000-hour remote run}; controlled from another continent.
    \item \gd{Assembly manual} validated by an undergrad with no electronics experience.
  \end{itemize}
  \vspace{0.3em}
  The cheapest buildable three-media selection cycler in the field. \\
  \textbf{The problems below are correctable --- framing and analysis, not hardware.}
\end{frame}

\begin{frame}{OpenEvo: two verifiable errors}
  \begin{block}{\bd{Error 1} --- the headline figure contradicts itself}
  Abstract: ``\bd{55\% faster}.'' \quad Results: ``4.9 vs 7.6 h, a \bd{36\% increase in growth rate}.'' \\
  Same data. $7.6\!\to\!4.9$\,h is a \emph{36\% shorter doubling time} \textbf{=} a \emph{55\% higher rate}
  ($1/4.9 \div 1/7.6 = 1.55$). The results sentence mislabels one as the other.
  \end{block}
  \begin{block}{\bd{Error 2} --- a false claim about a competitor}
  ``Pioreactor\dots only the \emph{software} is open source.'' \\
  False: the hardware is openly licensed (CC~BY-SA~4.0, per its \texttt{LICENSE}). \\
  \small\emph{(Our own first draft misnamed it CERN-OHL --- a contributor corrected it against the primary file.)}
  \end{block}
\end{frame}

\begin{frame}{OpenEvo: the deeper gap --- evidence}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \textbf{\bd{$n=1$}} \\[0.3em]
      One clone, one population sequenced. No replicates $\Rightarrow$ no convergence $\Rightarrow$
      \emph{cannot say which mutation caused the adaptation.} The LTEE runs \textbf{twelve} for this reason.
    \column{0.5\textwidth}
      \textbf{\bd{No archiving}} \\[0.3em]
      One hand-taken glycerol stock. The LTEE's real innovation was the \emph{freezer}, not the flask
      --- revive an ancestor, race it against a descendant. Without it, fitness is unmeasurable.
  \end{columns}
  \vspace{0.8em}
  \begin{center}
  \emph{OpenEvo cites Barrick's ``\dots Archiving Populations, and Measuring Fitness'' --- then implements neither.}
  \end{center}
\end{frame}

\begin{frame}{\texttt{devo} has no champion --- yet}
  OpenEvo is the best \emph{actuator}, not the champion: \ac{devo is defined by evidence, not actuation.}
  \vspace{0.3em}
  \begin{block}{What the actuator must connect to (only one is new hardware)}
  \begin{itemize}\small
    \item \textbf{replicate manifold} (borrow Pioreactor's cluster) $\to$ convergence
    \item \textbf{automated chilled archive} --- \emph{the one new module} $\to$ fossil record
    \item \textbf{sequencing $+$ \texttt{breseq}} $\to$ correct variant calls
    \item \textbf{fitness loop} (reuses the OD hardware) $\to$ selection coefficients
    \item \textbf{selection design} (growth-coupling) $\to$ dodge the ALE trap
  \end{itemize}
  \end{block}
  \begin{center}\small Champion $=$ mostly connective tissue. Keep the actuator cheap.\end{center}
\end{frame}

\begin{frame}{The proposal: build two, share the parts}
  \begin{columns}[T]
    \column{0.5\textwidth}
      \textbf{\textcolor{accent}{Characterisation machine}} \\
      \emph{Non-contact is inviolable.}
      \begin{itemize}\small
        \item + optode DO / pH (read through wall)
        \item + dispersive spectrometer 340--850\,nm
        \item + sealed gas-controlled lid
      \end{itemize}
    \column{0.5\textwidth}
      \textbf{\textcolor{violet}{Evolution machine}} \\
      \emph{Cheap $\times$ many; resist sensing bloat.}
      \begin{itemize}\small
        \item + 8 replicate chambers, one interface
        \item + \textbf{automated chilled archive} (nobody has built this)
        \item + batch-excursion mode; \texttt{breseq}
      \end{itemize}
  \end{columns}
  \vspace{0.7em}
  \begin{block}{Why two? The sensors sort themselves by contact.}
  Optical (DO/pH/spectrometry) $\to$ characterisation. Electrode (conductivity/ERT/viable-cell) $\to$ evolution.
  \emph{Share} the module footprint, sensor designs, firmware, calibration --- not the vessel.
  \end{block}
\end{frame}

\begin{frame}{Biopunk build I --- \textcolor{accent}{proto}: an updated Chi.Bio}
  \textbf{Chassis: Chi.Bio.} \emph{Design law: everything through the wall --- no immersed electrodes.}
  \begin{itemize}\small
    \item \textbf{graft} OpenEvo's closed, gas-capable vessel --- fixes the open-atmosphere gap
    \item \textbf{add} DO $+$ pH optodes (read through the wall) $+$ Hamamatsu C12880MA dispersive spectrometer
    \item \textbf{port} to Raspberry Pi (all-I\textsuperscript{2}C behind the \texttt{0x74} mux --- a respin, not a rewrite)
  \end{itemize}
  \vspace{0.2em}
  \begin{center}\small
  Pi port $\to$ gas vessel $\to$ optodes $\to$ spectrometer $\to$
  \ac{8 richly-instrumented, gas-controlled, closed reactors on one Pi.}
  \end{center}
\end{frame}

\begin{frame}{Biopunk build II --- \textcolor{violet}{devo}: evolution machine $+$ evidence}
  \textbf{Chassis: Pioreactor cluster} (or an OpenEvo base for teaching / shoestring). \\
  The champion-maker is the \emph{evidence apparatus} --- chassis-agnostic:
  \begin{itemize}\small
    \item \textbf{replicate fleet} ($\ge$8 lineages) $\to$ convergence
    \item \textbf{automated chilled archiver} --- \ac{the field's missing freezer; build it first}
    \item WGS $\to$ \texttt{breseq} $\to$ shared causal mutations; revive-ancestor fitness assay
    \item $+$ OpenEvo 3-media fluidics $+$ closed vessel; conductivity; batch-excursion
  \end{itemize}
  \vspace{0.2em}
  \begin{center}\small Prove it at n$\,\ge\,$8 with an archived record $=$ the \texttt{devo} champion $\to$ then the AI/CRISPR autopilot.\end{center}
\end{frame}

\begin{frame}{Biopunk build III --- \textcolor{warn}{bior}: Pioreactor $+$ add-ons}
  \textbf{Chassis: buy Pioreactor.} For \texttt{bior}, turnkey reliability \emph{is} the product --- mostly configuration, not fabrication.
  \begin{itemize}\small
    \item \textbf{add-on} the shared DO/pH optode module via the plugin path (same footprint --- no new board)
    \item \textbf{add-on} eVOLVER Smart Sleeves \emph{only} on the vials that truly need arbitrary per-vial control
    \item stand on its monthly-maintained software $+$ leader/worker cluster
  \end{itemize}
  \vspace{0.2em}
  \begin{center}\small Guardrail: don't over-graft --- the value of \texttt{bior} is that you \ac{didn't build it}.\end{center}
\end{frame}

\begin{frame}{These are teaching instruments too}
  \begin{itemize}
    \item \ac{EVE} --- eLife paper is literally ``\dots and its educational use''; $<$\$200; validated
      in a \emph{real classroom} evolving \emph{E.~coli} under chloramphenicol.
    \item \ac{Pioreactor} --- educators portal; one device teaches micro + EE + CS at once.
    \item \ac{OpenEvo} --- the manual \emph{is} a course; anchored a global hackathon.
    \item \ac{Chi.Bio} --- ``remove the lid, observe fluorescence by eye.''
  \end{itemize}
  \vspace{0.6em}
  \begin{center}
  \emph{But the teaching inherits the research blind spots} --- archiving, replicates, and
  carrying capacity are as under-taught as they are under-built.
  \end{center}
\end{frame}

\begin{frame}{Where this meets CRISPR}
  \begin{itemize}
    \item These tools \ac{evolve} CRISPR --- xCas9, base editors, compact Cas9, all PACE / eVOLVER (Hu '18; Thuronyi '19; Huang '22).
    \item CRISPR runs \ac{inside} these tools --- pooled screens in continuous culture; Cas-based growth-coupled selection.
    \item CRISPR may \ac{close the archiving gap} --- in-cell molecular recorders instead of the freezer.
  \end{itemize}
  \vspace{0.5em}
  \begin{center}\small The \texttt{bior} landmark is itself a CRISPR result --- the intersection is already here.\end{center}
\end{frame}

\begin{frame}{The goal: self-driving evolution}
  OpenEvo \emph{pitched} AI-directed evolution --- ``AI to supervise the evolution'' --- but built only the actuator.
  \begin{block}{The equivalence that makes it buildable}
  The observation layer an AI needs \ac{is} the evidence apparatus that makes a \texttt{devo} champion. \\
  \textbf{Apparatus $=$ champion; AI $=$ autopilot. Build the apparatus first, then close the loop.}
  \end{block}
  \begin{itemize}\small
    \item \textbf{observe}: growth, fitness, genotype (archive $\to$ \texttt{breseq}) $+$ in-cell CRISPR recorder
    \item \textbf{act}: media / light / dilution $+$ \ac{CRISPR-targeted mutation} (EvolvR --- an IGI tool)
    \item \textbf{learn}: rule-based $\to$ Bayesian opt $\to$ RL, across the replicate fleet
  \end{itemize}
  \begin{center}\small Prove it: AI-directed vs.\ fixed schedule, same target, replicate fleet. \quad\texttt{docs/ai-crispr-evolution-project.md}\end{center}
\end{frame}

\begin{frame}{Take-aways / call to build}
  \begin{enumerate}
    \item Stop comparing these as one contest --- \ac{three purposes}, different winners.
    \item Evolution has \bd{no champion}: nobody archives or replicates. \emph{That's the opening.}
    \item The killer feature nobody has shipped: an \ac{automated chilled fraction collector}.
    \item OpenEvo's errors are real but \gd{fixable} --- and we filed them, fairly, with fixes.
  \end{enumerate}
  \vspace{0.8em}

  \end{center}
\end{frame}
